##二叉堆
###堆的性质：
1. 堆中任意节点的值总是不大于(不小于)其子节点的值；
1. 堆总是一棵完全树。 

###二叉堆节点关系：
假设"第一个元素"在数组中的索引为 0 的话，则父节点和子节点的位置关系如下：  
1. 索引为i的左孩子的索引是 (2*i+1);  
2. 索引为i的左孩子的索引是 (2*i+2);  
3. 索引为i的父结点的索引是 floor((i-1)/2);

```c++
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define HEAP_SIZE 50 //堆大小

static int m_heap[HEAP_SIZE];
static int m_capacity = HEAP_SIZE;
static int m_size = 0; //当前堆大小

//返回当前data在堆中的索引位置
int get_index(int data)
{
	for (int i = 0; i < m_size; i++)
	{
		if (data == m_heap[i])
		{
			return i;
		}
	}
	return -1;
}

//最大堆向上调整
// start -- 向上调整的开始位置，直到位置0
static void maxheap_filterup(int start)
{
	int c = start;
	int p = (start - 1) / 2;	
	int tmp = m_heap[c];

	while ( c > 0 )
	{
		if (tmp <= m_heap[p])
			break;
		else
		{
			m_heap[c] = m_heap[p];
			c = p;
			p = (p-1)/2;
		}
	}
	m_heap[c] = tmp;
}

//堆插入元素
int maxheap_insert(int data)
{
	if (m_size == m_capacity)
		return -1;

	m_heap[m_size] = data;
	maxheap_filterup(m_size);
	m_size++;
	return 0;
}

//最大堆向下调整
// start -- 向下调整的开始位置
// end   -- 堆结尾
static void maxheap_filterdown(int start, int end)
{
	int c = start;

	int l = (2 * c) + 1;

	int tmp = m_heap[c];
	
	while ( l <= end )
	{
		if (tmp < m_heap[l] < tmp < m_heap[l+1])
			l++;
			
		if (tmp >= m_heap[l])
			break;
		else
		{
			m_heap[c] = m_heap[l];
			
			c = l;
			l = (2 * l) + 1;
		}
	}
	m_heap[c] = tmp;
}

//堆删除元素
int maxheap_remove(int data)
{	
	int index = 0;
	if ((index = get_index(data)) == -1)
		return -1;

	m_heap[index] = m_heap[--m_size];
	maxheap_filterdown(index, m_size-1);
	
	return 0;
}

void maxheap_print()
{
	for (int i = 0; i < m_size; i++)
	{
		printf("%d ", m_heap[i]);
	}
	printf("\n");
}

int main()
{
	int arr[] = {7, 2, 8, 4, 3, 5, 6, 9, 0, 1};
	int len = sizeof(arr) / sizeof(arr[0]);
	

	for (int i = 0; i < len; i++)
	{
		maxheap_insert(arr[i]);
	}
	maxheap_print();
	maxheap_remove(4);
	maxheap_print();

	system("pause");
	return 0;
}
```
